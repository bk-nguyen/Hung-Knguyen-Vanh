\begin{frame}{Bài 3}

\begin{center}
\textbf{Đường kính của cây}
\end{center}

{\small
Cho một cây gồm $n$ đỉnh và $n-1$ cạnh.

Hãy tìm độ dài đường đi dài nhất giữa hai đỉnh bất kỳ trong cây.
}

\vspace{0.3cm}

\textbf{Input}
\begin{itemize}
\item Số nguyên $n$
\item $n-1$ dòng mỗi dòng là cạnh $u, v$
\end{itemize}

\end{frame}

\begin{frame}{Bài 3}

\textbf{Output}
\begin{itemize}
\item Độ dài đường kính của cây
\end{itemize}

\vspace{0.3cm}

\textbf{Giới hạn}
\begin{itemize}
\item $1 \le n \le 2 \times 10^5$
\end{itemize}

\end{frame}

\begin{frame}{Bài 3}

\textbf{Ý tưởng}

\vspace{0.4cm}

\begin{itemize}
\item BFS từ một đỉnh bất kỳ → tìm đỉnh xa nhất $u$
\item BFS từ $u$ → tìm đỉnh xa nhất $v$
\end{itemize}

\vspace{0.3cm}

\textbf{Khoảng cách $u \rightarrow v$ chính là đường kính}

\end{frame}
\begin{frame}[fragile]{Bài 3 - Code}

\begin{lstlisting}
vector<int> adj[200005];
int dist[200005];

int bfs(int s){
    memset(dist,-1,sizeof(dist));
    queue<int> q;
    q.push(s);
    dist[s]=0;
    int last = s;
    while(!q.empty()){
        int u=q.front(); q.pop();
        last = u;

        for(int v:adj[u]){
            if(dist[v]==-1){
                dist[v]=dist[u]+1;
                q.push(v);
            }
        }
    }

    return last;
}
\end{lstlisting}

\end{frame}

\begin{frame}{Bài 4}

\begin{center}
\textbf{Đếm số lá}
\end{center}

{\small
Cho một cây gồm $n$ đỉnh.

Hãy đếm số đỉnh lá của cây (đỉnh có đúng 1 cạnh, trừ trường hợp $n=1$).
}

\vspace{0.3cm}

\textbf{Input}
\begin{itemize}
\item Số nguyên $n$
\item $n-1$ cạnh
\end{itemize}

\end{frame}

\begin{frame}{Bài 4}

\textbf{Output}
\begin{itemize}
\item Số lượng đỉnh lá
\end{itemize}

\vspace{0.3cm}

\textbf{Giới hạn}
\begin{itemize}
\item $1 \le n \le 2 \times 10^5$
\end{itemize}

\end{frame}

\begin{frame}{Bài 4}

\textbf{Ý tưởng}

\vspace{0.4cm}

\begin{itemize}
\item Đỉnh lá là đỉnh có bậc = 1
\item Duyệt tất cả các đỉnh
\item Đếm số đỉnh có degree = 1
\end{itemize}

\vspace{0.3cm}

\textbf{Lưu ý: nếu $n=1$ thì kết quả = 1}

\end{frame}

\begin{frame}[fragile]{Bài 4 - Code}

\begin{lstlisting}
int main(){
    int n;
    cin>>n;

    vector<int> deg(n+1,0);

    for(int i=1;i<n;i++){
        int u,v;
        cin>>u>>v;
        deg[u]++;
        deg[v]++;
    }
    if(n==1){
        cout<<1;
        return 0;
    }
    int cnt=0;
    for(int i=1;i<=n;i++){
        if(deg[i]==1) cnt++;
    }

    cout<<cnt;
}
\end{lstlisting}

\end{frame}
